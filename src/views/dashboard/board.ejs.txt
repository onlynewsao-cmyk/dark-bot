<script>
function filterRows(){
  const q = document.getElementById('search').value.toLowerCase();
  const st = document.getElementById('statusFilter').value;
  const cat = document.getElementById('catFilter').value;
  document.querySelectorAll('#nativeTable tbody tr').forEach(tr => {
    const okQ = tr.dataset.search.includes(q);
    const okS = st === 'all' || tr.dataset.status === st;
    const okC = cat === 'all' || tr.dataset.cat === cat;
    tr.style.display = okQ && okS && okC ? '' : 'none';
  });
}
async function toggleNative(name, enabled){
  const res = await fetch('/api/command-overrides/' + encodeURIComponent(name), {
    method:'PUT', headers:{'Content-Type':'application/json'}, body: JSON.stringify({ enabled })
  });
  if(!res.ok){ const e = await res.json().catch(()=>({error:'erro'})); return toast(e.error || 'Erro','error'); }
  location.reload();
}

const currentStyle = '<%= settings.menu_style || "classic" %>';
const styleSelect = document.getElementById('menuStyleSelect');
const names = ['Classic Dark','Cyber Matrix','Royal Neon','Shadow Box','Blade Clean','Arcane Lines','Terminal Web','Aurea Ego','Moonlight','Chaos Grid','Minimal Dark','Nova Premium'];
for(let i=0;i<120;i++){
  const id = i === 0 ? 'classic' : 'dark-' + i;
  const opt = document.createElement('option');
  opt.value = id;
  opt.textContent = `${String(i).padStart(3,'0')} — ${names[i % names.length]} ${['⚡','♾️','🌑','👑','🕸️','💎'][i%6]}`;
  if(id === currentStyle) opt.selected = true;
  styleSelect.appendChild(opt);
}
document.getElementById('menuStyleForm').addEventListener('submit', async e => {
  e.preventDefault();
  const data = Object.fromEntries(new FormData(e.target).entries());
  const res = await fetch('/api/settings/menu-style', { method:'POST', headers:{'Content-Type':'application/json'}, body: JSON.stringify(data) });
  if(res.ok) toast('Aparência salva','success'); else toast('Erro ao salvar aparência','error');
});

document.getElementById('menuMediaForm').addEventListener('submit', async e => {
  e.preventDefault();
  const fd = new FormData(e.target);
  const res = await fetch('/api/settings/menu-media', { method:'POST', body: fd });
  if(res.ok){ toast('Mídia do menu salva','success'); e.target.reset(); }
  else { const err = await res.json().catch(()=>({error:'erro'})); toast(err.error || 'Erro','error'); }
});
